94  #define OS_MsgReceive( pt, mbox, msg, timeout ) \
95  do \
96  { \
97  OS_Delay(ME, timeout); \
98  LC_SET((pt)->lc); \
99  if(!mbox->MBoxSem.SemCount && !(TaskTimerExpired(ME))) \
100  { \
101  OS_Suspend((K_LIST *)&mbox->MBoxSem); \
102  return PT_WAITING; \
103  } \
104  } \
105  while(0); \
106  if (sem.SemCount) \
107  { \
108  sem.SemCount--; \
109  msg = (OS_msg *)KQ_qdelete((K_LIST *)mbox) \
110  } \
