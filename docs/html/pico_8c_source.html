249 OS_Init( void )
250 {
251  BYTE index = 0;
252  /*
253  * initialize the target's tick hardware
254  */
255  SetupTickInterrupt();
256  k_ready_list.next = k_ready_list.last = &k_ready_list;

373 void
374 OS_ResumeTask( TCB_Entry *tcbp )
375 {
376  TCB_Entry *pReadyList;
377  /*
378  * remove the task from any queue it's waiting on
379  * insert it onto the ready queue
380  */
381  KQ_ndelete( (K_LIST *)tcbp );
382  pReadyList = (TCB_Entry *)k_ready_list.next;
383  while( &k_ready_list != (K_LIST *)pReadyList )
384  {
385  if((pReadyList->Flags & PRIOMASK) > (tcbp->Flags & PRIOMASK))
386  {
387  break;
388  }
389  pReadyList = (TCB_Entry *)pReadyList->TcbLink.next;
390  }
391  KQ_qinsert((K_LIST *)pReadyList->TcbLink.last, (K_LIST *)tcbp);
392 }

903 void
904 ServiceOSTimers( void )
905 {
906  BYTE TcbIndex;
907  TIMER_T temp_t;
908  TIMER_T ElapsedTime;
909  temp_t = getSysTick();
910  ElapsedTime = (temp_t - LastTick);
911  LastTick = temp_t;
912  if (0 != ElapsedTime)
913  {
914  #ifdef USES_UIP
915  if ( UIP_Timer > ElapsedTime )
916  {
917  UIP_Timer -= ElapsedTime;
918  }
919  else
920  {
921  UIP_Timer = 0;
922  }
923  if ( ARP_Timer > ElapsedTime )
924  {
925  ARP_Timer -= ElapsedTime;
926  }
927  else
928  {
929  ARP_Timer = 0;
930  }
931  #endif
932  for (TcbIndex = 0; TcbIndex < N_TASKS; TcbIndex++)
933  {
934  if ((0 != TCB[TcbIndex].gpTimer) && (NO_TIMEOUT != TCB[TcbIndex].gpTimer))
935  {
936  if ( TCB[TcbIndex].gpTimer > ElapsedTime )
937  {
938  TCB[TcbIndex].gpTimer -= ElapsedTime;
939  }
940  else
941  {
942  TCB[TcbIndex].gpTimer = TIME_EXPIRED;
943  }
944  }
945  if ( TCB_TIMING == (TCB[TcbIndex].Flags & TCB_TIMINGMASK))
946  {
947  if ( TCB[TcbIndex].Timer > ElapsedTime )
948  {
949  TCB[TcbIndex].Timer -= ElapsedTime;
950  }
951  else
952  {
953  TCB[TcbIndex].Timer = TIME_EXPIRED;
954  TCB[TcbIndex].Flags &= ~TCB_TIMING;
955  TCB[TcbIndex].Flags |= TCB_TIMEOUT;
956  KQ_ndelete((K_LIST *)&TCB[TcbIndex]);
957  OS_ResumeTask(&TCB[TcbIndex]);
958  }
959  }
960  }
961  }
